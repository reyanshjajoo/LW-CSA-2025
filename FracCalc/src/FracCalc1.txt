    // e.g. return ==> "1_1/4"
    public static String produceAnswer(String input) {
        // Break the input into three parts: first operand, operator, second operand
        String[] parts = input.trim().split("\\s+");
        String firstOperand = "";
        String operator = "";
        String secondOperand = "";
        if (parts.length >= 3) {
            firstOperand = parts[0];
            operator = parts[1];
            secondOperand = parts[2];
        } else if (parts.length == 2) {
            firstOperand = parts[0];
            operator = parts[1];
            secondOperand = "";
        }
        return secondOperand;
    }
